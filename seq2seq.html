class Solution(object):
    def divide(self, dividend, divisor):
        """
        :type dividend: int
        :type divisor: int
        :rtype: int
        """
        if divisor==0:
            return MAX_INT
        positive = (dividend < 0) is (divisor < 0)
        dividend, divisor = abs(dividend), abs(divisor)
        tmp=0
        while dividend >=divisor :
            dividend-=divisor
            tmp+=1
        if not positive:
            tmp = -tmp
        return tmp
